# frozen_string_literal: true

require 'rails_helper'

context 'when updating a brand' do
  context 'when the update is successful' do
    before do
      allow(controller).to receive(:authorize).with(brand)
      allow(brand).to receive(:update).with(brand_params).and_return(true)
      controller.instance_variable_set(:@brand, brand)
    end

    it 'returns status :ok and the brand JSON' do
      put :update, params: { id: brand.id, brand: brand_params }
      
      expect(response).to have_http_status(:ok)
      expect(response.body).to eq(brand.to_json)
    end
  end

  context 'when update fails due to validation errors' do
    before do
      allow(controller).to receive(:authorize).with(brand)
      allow(brand).to receive(:update).with(brand_params).and_return(false)
      allow(brand.errors).to receive(:full_messages).and_return(['Name is invalid'])
      controller.instance_variable_set(:@brand, brand)
    end

    it 'returns status :unprocessable_content and the error list' do
      put :update, params: { id: brand.id, brand: brand_params }
      
      expect(response).to have_http_status(:unprocessable_content)
      expect(JSON.parse(response.body)).to eq({ 'errors' => ['Name is invalid'] })
    end
  end

  context 'when a database uniqueness error occurs (ActiveRecord::RecordNotUnique)' do
    before do
      allow(controller).to receive(:authorize).with(brand)
      allow(brand).to receive(:update).with(brand_params).and_raise(ActiveRecord::RecordNotUnique)
      controller.instance_variable_set(:@brand, brand)
    end

    it 'returns status :unprocessable_content and specific subdomain error message' do
      put :update, params: { id: brand.id, brand: brand_params }
      
      expect(response).to have_http_status(:unprocessable_content)
      expect(JSON.parse(response.body)).to eq({ 'errors' => ['Subdomain is already taken'] })
    end
  end
end

context 'when getting current_user' do
  context 'when current_user is already cached in @current_user variable' do
    it 'returns the existing user without re-decoding' do
      controller.instance_variable_set(:@current_user, user)
      
      expect(controller.current_user).to eq(user)
    end
  end

  context 'when the Authorization header is missing' do
    before do
      request.headers['Authorization'] = nil
    end

    it 'returns nil' do
      expect(controller.current_user).to be_nil
    end
  end

  context 'when a valid token is provided and decodes as an Array' do
    before do
      request.headers['Authorization'] = 'Bearer valid_token'
      allow(AuthHelper).to receive(:decode).with('valid_token').and_return([{ 'user_id' => user.id }])
    end

    it 'finds and returns the current user' do
      expect(controller.current_user).to eq(user)
    end
  end

  context 'when a valid token is provided and decodes as a Hash' do
    before do
      request.headers['Authorization'] = 'Bearer valid_token'
      allow(AuthHelper).to receive(:decode).with('valid_token').and_return({ user_id: user.id })
    end

    it 'finds and returns the current user' do
      expect(controller.current_user).to eq(user)
    end
  end

  context 'when an error occurs during decoding or finding (StandardError)' do
    before do
      request.headers['Authorization'] = 'Bearer invalid_token'
      allow(AuthHelper).to receive(:decode).with('invalid_token').and_raise(StandardError.new('Fail'))
    end

    it 'rescues the error and returns nil' do
      expect(controller.current_user).to be_nil
    end
  end
end
